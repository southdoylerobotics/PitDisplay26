// --- CONFIGURATION --- const TEAM_NUMBER = 3824; const AVERAGES_URL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vSoFDHMJqYQfAOO-7AcFcX9VXyuxIZQ9AuJ21BLhdIoZA-mXa_B6NijbOj0ZWmK_SQlt-gEXMf21ei_/pub?gid=131746992&single=true&output=csv"; document.getElementById('blue-prob-text').innerText = (100 - rProb) + "%"; renderCharts(match.redTeams, match.blueTeams, rD, bD); renderBadges(match.redTeams, match.blueTeams); } // PERCENTAGE HELPER: Handles 0.75 vs 75 correctly const toPct = (v) => { let val = parseFloat(v) || 0; return (val <= 1.1) ? (val * 100).toFixed(0) : val.toFixed(0); }; function renderCharts(rLabs, bLabs, rD, bD) { Object.values(charts).forEach(c => c.destroy()); const sum = (d, k) => d.reduce((s, t) => s + (parseFloat(t[k]) || 0), 0); T-${n}
${rnk}
`; }); }; draw(red, 'red-citrus'); draw(blue, 'blue-citrus'); } // Wrapper for the team lookup used in badges function getT(num) { const team = teamDataCache.find(t => Number(t['Team Number'] || t['team_number'] || t['Team']) === Number(num)); return team || {"Avg Citrus": 1}; } function dismissWarning() { dismissedMatch = Number(document.getElementById('match-selector').value); document.getElementById('live-banner').classList.remove('queueing'); } // Initial Call fetchData(); setInterval(fetchData, 30000); // Update selector to use handleManualSelection document.getElementById('match-selector').setAttribute('onchange', 'handleManualSelection()');
